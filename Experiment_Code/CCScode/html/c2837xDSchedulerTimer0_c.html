146void configureTimer0(float baseRate, float systemClock)



147{



148    #ifdef CPU1



149        EINT;



150        uint32_t ipcCountVar = 0;



151        if (0x425 == mwInitStart)



152        {



153            /* Wait until completion of CPU2 GPIO initialization */



154            while ((0x36E != mwInitEnd))



155            {



156                ipcCountVar++;



157            }



158        }



159    DINT;



160    #else



161	uint32_t *pulMsgRam;



162	pulMsgRam = (void *)CPU01_TO_CPU02_PASSMSG;



163        IPCLiteLtoRDataWrite(IPC_FLAG0, pulMsgRam[2], 0x36E,



164                         IPC_LENGTH_32_BITS, IPC_FLAG31);



165    #endif



166        



167    /* Initialize CPU Timers here only if Profiling is disabled.



168     * Because if profiling is enabled, CPU Timers are initialized in 



169     * MW_c28xx_board.c in the generated code.



170     */



171	#ifndef MW_EXEC_PROFILER_ON



172	//



173    // CPU Timer 0



174    // Initialize address pointers to respective timer registers:



175    //



176    CpuTimer0.RegsAddr = &CpuTimer0Regs;



177



178    //



179    // Initialize timer period to maximum:



180    //



181    CpuTimer0Regs.PRD.all  = 0xFFFFFFFF;



182



183    //



184    // Initialize pre-scale counter to divide by 1 (SYSCLKOUT):



185    //



186    CpuTimer0Regs.TPR.all  = 0;



187    CpuTimer0Regs.TPRH.all = 0;



188



189    //



190    // Make sure timer is stopped:



191    //



192    CpuTimer0Regs.TCR.bit.TSS = 1;



193



194    //



195    // Reload all counter register with period value:



196    //



197    CpuTimer0Regs.TCR.bit.TRB = 1;



198



199    //



200    // Reset interrupt counters:



201    //



202    CpuTimer0.InterruptCount = 0;



203	#endif



204



205	/* Configure CPU-Timer 0 to interrupt every 0.0025 sec. */



206	/* Parameters:  Timer Pointer, CPU Freq in MHz, Period in usec. */



207	ConfigCpuTimer(&CpuTimer0, systemClock, baseRate * 1000000);



208	StartCpuTimer0();



209



210	EALLOW;



211	PieVectTable.TIMER0_INT = &TINT0_isr;     /* Hook interrupt to the ISR*/



212	EDIS;



213



214	PieCtrlRegs.PIEIER1.bit.INTx7 = 1;   /* Enable interrupt TINT0 */



215	IER |= M_INT1;



216}

218void enableTimer0Interrupt(void)



219{



220	asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */



221	PieCtrlRegs.PIEIER1.all |= (1 << 6); /* Enable PIER bit for timer 0 */



222	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



223}



224



225bool disableTimer0Interrupt(void)



226{



227    bool isrEnabled = false;



228	



229    asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */



230    



231    isrEnabled = PieCtrlRegs.PIEIER1.all & (1 << 6) ;



232    



233	PieCtrlRegs.PIEIER1.all &= ~(1 << 6); /* Disable PIER bit for timer 0 */



234	asm(" RPT #5 || NOP");



235	IFR &= 0xFFFE;



236	PieCtrlRegs.PIEACK.all = 0x1;



237	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



238    



239    return isrEnabled;



240}



241



242void restoreTimer0Interrupt(bool isrStatus){



243    if(isrStatus)



244        enableTimer0Interrupt();



245    else



246        disableTimer0Interrupt();



247}
